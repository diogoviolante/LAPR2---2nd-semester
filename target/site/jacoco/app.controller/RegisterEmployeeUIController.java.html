    /**

    public String generateID(String name) {
        int max = 100000;
        String empID = "";


        count = count % max;
        String numEmp = String.valueOf(count);
        while (numEmp.length() < 5) {
            numEmp = "0" + numEmp;
        }


        String[] id = name.split(" ");
        for (String s : id) {
            empID = empID + String.valueOf(s.toUpperCase().charAt(0));
        }
        empID = empID + numEmp;

        count++;
        System.out.println("Name Id: " + empID);
        return empID;
    }
